# Installation
```
npm i ui-kir-on
```
# Getting started with kir ui
```
import * as React from 'react';
import ReactDOM from 'react-dom';
import Button from 'ui-kir-on';

function App() {
return (
<Button color="teal">
Hello World
</Button>
);
}
```
